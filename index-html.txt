const PKEY = 'imamio-profiles', CKEY = 'imamio-current';
let profiles = [], CUR = null, S = { events: [], tx: [], notes: [] };

function lsGet(k, def) { try { const r = localStorage.getItem(k); return r ? JSON.parse(r) : def; } catch (e) { return def; } }
function lsSet(k, v) { try { localStorage.setItem(k, JSON.stringify(v)); } catch (e) { alert('Не удалось сохранить данные'); } }

profiles = lsGet(PKEY, []);
/* миграция со старой версии */
(function () {
  const old = lsGet('imamio-data-v1', null);
  if (old && profiles.length === 0) {
    const id = uid();
    profiles = [{ id, name: 'Основной', pin: null }];
    lsSet(PKEY, profiles);
    lsSet('imamio-data-' + id, old);
    localStorage.removeItem('imamio-data-v1');
  }
})();
CUR = lsGet(CKEY, null);
if (CUR && !profiles.find(p => p.id === CUR)) CUR = null;

function loadProfile(id) {
  CUR = id;
  lsSet(CKEY, id);
  S = Object.assign({ events: [], tx: [], notes: [] }, lsGet('imamio-data-' + id, {}));
  renderApp();
  startReminderLoop();
}
function save() { if (CUR) lsSet('imamio-data-' + CUR, S); }
function hashPin(p) { let h = 5381; for (const c of p) h = ((h << 5) + h + c.charCodeAt(0)) | 0; return String(h); }

/* ================== УТИЛИТЫ ================== */
function uid() { return Date.now().toString(36) + Math.random().toString(36).slice(2, 7); }

    h += `<div class="card form">

function createProfile() {
  const name = document.getElementById('np-name').value.trim();
  const pin = document.getElementById('np-pin').value.trim();
  if (!name) return;
  if (pin && !/^\d{4}$/.test(pin)) { alert('PIN — ровно 4 цифры'); return; }
  const id = uid();
  profiles.push({ id, name, pin: pin ? hashPin(pin) : null });
  lsSet(PKEY, profiles);
  newProfileOpen = false;
  loadProfile(id);
}
function tryOpen(id) {
  const p = profiles.find(p => p.id === id);
  if (p.pin) {
    const entered = prompt('PIN для профиля \u00AB' + p.name + '\u00BB:');
    if (entered === null) return;
    if (hashPin(entered.trim()) !== p.pin) { alert('Неверный PIN'); return; }
  }
  loadProfile(id);
}
function switchProfile() { CUR = null; lsSet(CKEY, null); renderApp(); }

/* ================== СЕГОДНЯ ================== */
function rToday() {
  const T = todayStr();
  const overdue = S.events.filter(e => !e.done && e.date < T);
  const today = S.events.filter(e => e.date === T && !e.done).sort((a, b) => (a.time || '99').localeCompare(b.time || '99'));
  const week = S.events.filter(e => !e.done && e.date > T && daysBetween(T, e.date) <= 7)
    .sort((a, b) => a.date.localeCompare(b.date) || (a.time || '99').localeCompare(b.time || '99'));
  const mtx = S.tx.filter(t => mKey(t.date) === mKey(T));
  const inc = mtx.filter(t => t.type === 'income').reduce((s, t) => s + t.amount, 0);
  const exp = mtx.filter(t => t.type === 'expense').reduce((s, t) => s + t.amount, 0);
  let h = '';
  if (overdue.length) {
    h += `<section class="card alert"><div class="card-h red">Просрочено \u00B7 ${overdue.length}</div>` +
      overdue.slice(0, 5).map(e => evRow(e, { date: true })).join('') +
      (overdue.length > 5 ? `<button class="link" onclick="setTab('plan')">показать все \u2192</button>` : '') + '</section>';
  }
  h += `<section class="card"><div class="card-h">План на сегодня</div>` +
    (today.length ? today.map(e => evRow(e)).join('') : `<div class="empty">На сегодня ничего не запланировано.</div>`) + '</section>';
  if (week.length) {
    h += `<section class="card"><div class="card-h">Ближайшие 7 дней</div>` + week.slice(0, 6).map(e => evRow(e, { date: true })).join('') + '</section>';
  }
  h += `<section class="card"><div class="card-h">Финансы за месяц</div>
    <div class="fin-grid">
      <div><div class="fin-l">Доход</div><div class="fin-v green">${money(inc)}</div></div>
      <div><div class="fin-l">Расход</div><div class="fin-v red">${money(exp)}</div></div>
      <div><div class="fin-l">Баланс</div><div class="fin-v">${money(inc - exp)}</div></div>
    </div>
    <button class="link" onclick="setTab('money')">подробнее \u2192</button></section>`;
  return h;
}
function daysBetween(a, b) { return Math.round((new Date(b) - new Date(a)) / 86400000); }

function rCal() {
  const T = todayStr();
  const first = new Date(calY, calM, 1);
  const startShift = (first.getDay() + 6) % 7; /* Пн = 0 */
  const daysInM = new Date(calY, calM + 1, 0).getDate();
  const daysPrev = new Date(calY, calM, 0).getDate();
  const evByDate = {};
  for (const e of S.events) { if (!e.done) (evByDate[e.date] = evByDate[e.date] || []).push(e); }

  let cells = '';
  const total = Math.ceil((startShift + daysInM) / 7) * 7;
  for (let i = 0; i < total; i++) {
    const dayNum = i - startShift + 1;
    let ds, label, other = false;
    if (dayNum < 1) { label = daysPrev + dayNum; other = true; ds = null; }
    else if (dayNum > daysInM) { label = dayNum - daysInM; other = true; ds = null; }
    else {
      label = dayNum;
      ds = calY + '-' + String(calM + 1).padStart(2, '0') + '-' + String(dayNum).padStart(2, '0');
    }
    if (!ds) { cells += `<button class="cal-cell other" disabled>${label}</button>`; continue; }
    const evs = evByDate[ds] || [];
    const dots = evs.slice(0, 3).map(e => `<div class="cal-dot" style="background:${(TYPES[e.type] || TYPES.task).color}"></div>`).join('');
    cells += `<button class="cal-cell${ds === T ? ' today' : ''}${ds === calSel ? ' sel' : ''}" onclick="calSel='${ds}';renderApp()">
      ${label}<div class="cal-dots">${dots}</div></button>`;
  }

  const selEvs = (S.events.filter(e => e.date === calSel)).sort((a, b) => (a.time || '99').localeCompare(b.time || '99'));
  const yearCount = S.events.filter(e => !e.done && e.date.startsWith(String(calY))).length;

  let h = `<div class="month-nav">
    <button class="btn-ghost" onclick="calShift(-12)">\u00AB</button>
    <button class="btn-ghost" onclick="calShift(-1)">\u2190</button>
    <div class="month-name">${MONTHS[calM]} ${calY}</div>
    <button class="btn-ghost" onclick="calShift(1)">\u2192</button>
    <button class="btn-ghost" onclick="calShift(12)">\u00BB</button>
  </div>
  <section class="card">
    <div class="cal-grid">${['Пн','Вт','Ср','Чт','Пт','Сб','Вс'].map(w => `<div class="cal-wd">${w}</div>`).join('')}${cells}</div>
    <div class="empty" style="padding-bottom:0">Запланировано на ${calY} год: ${yearCount}</div>
  </section>`;

  h += evFormOpen ? evForm(calSel) :
    `<button class="btn-main" onclick="evFormOpen=true;evEditId=null;renderApp();setTimeout(()=>{var i=document.getElementById('ev-title');if(i)i.focus()},50)">+ Добавить на ${dHuman(calSel)}</button>`;

  h += `<section class="card"><div class="card-h">${dHuman(calSel)}</div>` +
    (selEvs.length ? selEvs.map(e => evRow(e, { del: true, edit: true })).join('') : `<div class="empty">На этот день ничего нет.</div>`) + '</section>';
  return h;
}
function calShift(d) {
  const nd = new Date(calY, calM + d, 1);
  calY = nd.getFullYear(); calM = nd.getMonth();
  renderApp();
}

function sums(list) {
  return {
    inc: list.filter(t => t.type === 'income').reduce((s, t) => s + t.amount, 0),
    exp: list.filter(t => t.type === 'expense').reduce((s, t) => s + t.amount, 0)
  };
}

function rMoneyMonth() {

  const mtx = S.tx.filter(t => mKey(t.date) === MK).sort((a, b) => b.date.localeCompare(a.date));
  const { inc, exp } = sums(mtx);
  const cats = {};
  mtx.filter(t => t.type === 'expense').forEach(t => { cats[t.category] = (cats[t.category] || 0) + t.amount; });
  const byCat = Object.entries(cats).sort((a, b) => b[1] - a[1]);

  let h = `<div class="month-nav">
    <button class="btn-ghost" onclick="shiftMonth(-1)">\u2190</button>

        <div class="bar-track"><div class="bar-fill" style="width:${Math.max(3, s / byCat[0][1] * 100)}%"></div></div>
      </div>`).join('') + '</section>';
  }

  h += `<section class="card"><div class="card-h">Операции</div>` +
    (mtx.length ? mtx.map(t => `<div class="tx">

function rMoneyYear() {
  const ytx = S.tx.filter(t => t.date.startsWith(String(finYear)));
  const { inc, exp } = sums(ytx);
  let maxV = 0;
  const rows = [];
  for (let m = 0; m < 12; m++) {
    const mk = finYear + '-' + String(m + 1).padStart(2, '0');
    const s = sums(S.tx.filter(t => mKey(t.date) === mk));
    maxV = Math.max(maxV, s.inc, s.exp);
    rows.push([m, s]);
  }
  let h = `<div class="month-nav">
    <button class="btn-ghost" onclick="finYear--;renderApp()">\u2190</button>
    <div class="month-name">${finYear} год</div>
    <button class="btn-ghost" onclick="finYear++;renderApp()">\u2192</button>
  </div>
  <section class="card">${finGrid(inc, exp)}</section>
  <section class="card"><div class="card-h">По месяцам</div>`;
  if (!maxV) h += `<div class="empty">За этот год записей нет.</div>`;
  else {
    for (const [m, s] of rows) {
      if (!s.inc && !s.exp) continue;
      h += `<div class="bar-row">
        <div class="bar-top"><span>${MONTHS[m]}</span><span class="bar-sum">${money(s.inc - s.exp)}</span></div>
        <div class="bar-track"><div class="bar-fill g" style="width:${Math.max(2, s.inc / maxV * 100)}%"></div></div>
        <div class="bar-track" style="margin-top:2px"><div class="bar-fill" style="width:${Math.max(2, s.exp / maxV * 100)}%"></div></div>
      </div>`;
    }
    h += `<div class="empty" style="padding-bottom:0">зелёная — доход, красная — расход</div>`;
  }
  h += '</section>';

  const cats = {};
  ytx.filter(t => t.type === 'expense').forEach(t => { cats[t.category] = (cats[t.category] || 0) + t.amount; });
  const byCat = Object.entries(cats).sort((a, b) => b[1] - a[1]);
  if (byCat.length) {
    h += `<section class="card"><div class="card-h">Расходы за год по категориям</div>` +
      byCat.map(([c, s]) => `<div class="bar-row">
        <div class="bar-top"><span>${esc(c)}</span><span class="bar-sum">${money(s)}</span></div>
        <div class="bar-track"><div class="bar-fill" style="width:${Math.max(3, s / byCat[0][1] * 100)}%"></div></div>
      </div>`).join('') + '</section>';
  }
  return h;
}

function rMoneyAll() {

  }
  return h;
}

function askNotify(silent) {
  if (!('Notification' in window)) { if (!silent) alert('Браузер не поддерживает уведомления'); return; }
  if (Notification.permission === 'granted') { if (!silent) alert('Уведомления уже разрешены'); return; }
  Notification.requestPermission().then(p => {
    if (!silent) alert(p === 'granted' ? 'Уведомления включены' : 'Уведомления не разрешены');
  });
}
let remTimer = null;
function startReminderLoop() {
  if (remTimer) clearInterval(remTimer);
  checkReminders();
  remTimer = setInterval(checkReminders, 30000);
}
function checkReminders() {
  if (!CUR) return;
  const now = new Date();
  let changed = false;
  for (const e of S.events) {
    if (e.done || e.remind == null || !e.time || e.notified) continue;
    const [h, m] = e.time.split(':').map(Number);
    const [Y, Mo, D] = e.date.split('-').map(Number);
    const evAt = new Date(Y, Mo - 1, D, h, m);
    const notifyAt = new Date(evAt.getTime() - e.remind * 60000);
    if (now >= notifyAt && now <= new Date(evAt.getTime() + 5 * 60000)) {
      fireNotify(e, evAt);
      e.notified = true;
      changed = true;
    }
  }
  if (changed) save();
}
function fireNotify(e, evAt) {
  const body = (TYPES[e.type] || TYPES.task).label + ' \u00B7 ' + e.time + ', ' + dHuman(e.date);
  if ('Notification' in window && Notification.permission === 'granted') {
    try { new Notification('Imamio: ' + e.title, { body, icon: 'icon-192.png' }); return; } catch (err) {}
    if (navigator.serviceWorker && navigator.serviceWorker.ready) {
      navigator.serviceWorker.ready.then(reg => reg.showNotification('Imamio: ' + e.title, { body, icon: 'icon-192.png' })).catch(() => {});
      return;
    }
  }
  alert('\u23F0 ' + e.title + '\n' + body);
}

/* ---- экспорт события в календарь телефона (.ics) ---- */
function downloadICS(id) {
  const e = S.events.find(x => x.id === id);
  if (!e || !e.time) return;
  const [Y, Mo, D] = e.date.split('-').map(Number);
  const [h, m] = e.time.split(':').map(Number);
  const pad = n => String(n).padStart(2, '0');
  const dt = `${Y}${pad(Mo)}${pad(D)}T${pad(h)}${pad(m)}00`;
  const end = new Date(Y, Mo - 1, D, h + 1, m);
  const dtEnd = `${end.getFullYear()}${pad(end.getMonth() + 1)}${pad(end.getDate())}T${pad(end.getHours())}${pad(end.getMinutes())}00`;
  const alarm = e.remind != null ? `BEGIN:VALARM\r\nTRIGGER:-PT${e.remind || 1}M\r\nACTION:DISPLAY\r\nDESCRIPTION:${e.title}\r\nEND:VALARM\r\n` : '';
  const ics = `BEGIN:VCALENDAR\r\nVERSION:2.0\r\nPRODID:-//Imamio//RU\r\nBEGIN:VEVENT\r\nUID:${e.id}@imamio\r\nDTSTART:${dt}\r\nDTEND:${dtEnd}\r\nSUMMARY:${e.title.replace(/[\r\n]/g, ' ')}\r\n${alarm}END:VEVENT\r\nEND:VCALENDAR`;
  const blob = new Blob([ics], { type: 'text/calendar' });
  const a = document.createElement('a');
  a.href = URL.createObjectURL(blob);
  a.download = 'imamio-event.ics';
  a.click();
}

function importData(ev) {
  const file = ev.target.files[0];
  if (!file) return;
  const r = new FileReader();
  r.onload = () => {
    try {
      const d = JSON.parse(r.result);
      if (d && Array.isArray(d.events)) { S = Object.assign({ events: [], tx: [], notes: [] }, d); save(); renderApp(); alert('Данные восстановлены'); }
      else alert('Неверный формат файла');
    } catch (e) { alert('Не удалось прочитать файл'); }
  };
  r.readAsText(file);
  ev.target.value = '';
}
